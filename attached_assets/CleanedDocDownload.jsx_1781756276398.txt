// CleanedDocDownload.jsx
// Drop this into frontend/src/components/
// Usage: <CleanedDocDownload documentId={doc.id} />
//
// Shows a "Clean & Download" button on the Review page.
// Flow: POST /clean → show preview thumbnail → Download button appears.

import { useState } from "react";

export default function CleanedDocDownload({ documentId }) {
  const [state, setState] = useState("idle"); // idle | cleaning | ready | error
  const [preview, setPreview] = useState(null);
  const [downloadUrl, setDownloadUrl] = useState(null);
  const [error, setError] = useState(null);

  const API = import.meta.env.VITE_API_URL || "";

  async function handleClean() {
    setState("cleaning");
    setError(null);
    try {
      // Step 1: trigger cleaning pipeline
      const cleanRes = await fetch(`${API}/api/documents/${documentId}/clean`, {
        method: "POST",
      });
      if (!cleanRes.ok) throw new Error("Cleaning failed");

      // Step 2: fetch preview thumbnail
      const previewRes = await fetch(
        `${API}/api/documents/${documentId}/clean/preview`,
        { method: "POST" }
      );
      if (!previewRes.ok) throw new Error("Preview failed");
      const data = await previewRes.json();

      setPreview(data.preview_b64);
      setDownloadUrl(`${API}${data.download_url}`);
      setState("ready");
    } catch (e) {
      setError(e.message);
      setState("error");
    }
  }

  return (
    <div className="mt-6 rounded-xl border border-slate-200 bg-white p-4 shadow-sm">
      <h3 className="mb-3 text-sm font-semibold text-slate-700">
        Cleaned Document
      </h3>

      {/* Clean button — idle or error state */}
      {(state === "idle" || state === "error") && (
        <button
          onClick={handleClean}
          className="flex items-center gap-2 rounded-lg bg-teal-600 px-4 py-2
                     text-sm font-medium text-white hover:bg-teal-700
                     active:scale-95 transition-all"
        >
          <span>✨</span>
          Whiten Background & Clean
        </button>
      )}

      {/* Cleaning in progress */}
      {state === "cleaning" && (
        <div className="flex items-center gap-3 text-sm text-slate-500">
          <svg className="h-5 w-5 animate-spin text-teal-600"
               viewBox="0 0 24 24" fill="none">
            <circle className="opacity-25" cx="12" cy="12" r="10"
                    stroke="currentColor" strokeWidth="4" />
            <path className="opacity-75" fill="currentColor"
                  d="M4 12a8 8 0 018-8v8H4z" />
          </svg>
          Cleaning document — whitening background…
        </div>
      )}

      {/* Error */}
      {state === "error" && (
        <p className="mt-2 text-xs text-red-500">{error} — try again</p>
      )}

      {/* Preview + download — ready state */}
      {state === "ready" && (
        <div className="mt-2 space-y-3">
          {/* Thumbnail preview */}
          {preview && (
            <div className="overflow-hidden rounded-lg border border-slate-200">
              <img
                src={preview}
                alt="Cleaned document preview"
                className="w-full object-contain"
              />
            </div>
          )}

          {/* Action buttons */}
          <div className="flex flex-wrap gap-2">
            {/* Download PNG */}
            <a
              href={downloadUrl}
              download
              className="flex items-center gap-2 rounded-lg bg-indigo-600
                         px-4 py-2 text-sm font-medium text-white
                         hover:bg-indigo-700 active:scale-95 transition-all"
            >
              <span>⬇</span>
              Download Cleaned PNG
            </a>

            {/* Re-clean button */}
            <button
              onClick={() => { setState("idle"); setPreview(null); }}
              className="rounded-lg border border-slate-300 px-4 py-2
                         text-sm font-medium text-slate-600
                         hover:bg-slate-50 active:scale-95 transition-all"
            >
              Re-clean
            </button>
          </div>

          {/* Phase 2 nudge */}
          <p className="text-xs text-slate-400">
            Need to fix a specific region?{" "}
            <span className="text-teal-600 font-medium">
              Brush select + manual levels coming soon.
            </span>
          </p>
        </div>
      )}
    </div>
  );
}
